% Ability scores: "stats: [12, 8, 13, 10, 14, 15]" → six columns "12 (+1)".
% Modifier = floor((score-10)/2), exact for odd and even scores.
\newcommand{\dnd@mod}[1]{\the\numexpr(#1-\ifodd\numexpr#1\relax 11\else 10\fi)/2\relax}
\newcommand{\dnd@statcell}[1]{%
  \the\numexpr#1\relax\
  (\ifnum\numexpr(#1-\ifodd\numexpr#1\relax 11\else 10\fi)/2\relax<0\else+\fi\dnd@mod{#1})}
\def\dnd@splitstats#1,#2,#3,#4,#5,#6\dnd@endstats{%
  \begin{tabular*}{\linewidth}{@{\extracolsep{\fill}}cccccc@{}}
    \textbf{STR}&\textbf{DEX}&\textbf{CON}&\textbf{INT}&\textbf{WIS}&\textbf{CHA}\\
    \dnd@statcell{#1}&\dnd@statcell{#2}&\dnd@statcell{#3}&%
    \dnd@statcell{#4}&\dnd@statcell{#5}&\dnd@statcell{#6}%
  \end{tabular*}}

% Monster stat block — fed by a ```statblock code block (declare via nope-blocks:
% [statblock]). Keys follow the Fantasy Statblock Obsidian plugin, so the same
% block renders live in Obsidian: name, size, type, subtype, alignment, ac, hp,
% hit_dice, speed, stats, senses, languages, cr, damage_* and condition_immunities,
% plus the lists traits, actions, reactions and legendary_actions (name/desc items).
\newtcolorbox{dnd@statbox}{breakable,colback=DndMonsterTan,colframe=DndRuleRed,
  boxrule=0pt,toprule=2.5pt,bottomrule=2.5pt,arc=0pt,left=6pt,right=6pt}
\newenvironment{statblock}{%
  \begin{dnd@statbox}\dndsfbody\small
  \noindent{\normalfont\color{DndDarkRed}\scshape\LARGE\dnd@key{name}\par}
  \noindent{\itshape
    \dnd@ifkey{size}{\dnd@key{size} }%
    \dnd@ifkey{type}{\dnd@key{type}}%
    \dnd@ifkey{subtype}{ (\dnd@key{subtype})}%
    \dnd@ifkey{alignment}{, \dnd@key{alignment}}\par}
  \dnd@rule
  {\color{DndDarkRed}
  \dnd@ifkey{ac}{\par\noindent\textbf{Armor Class} \dnd@key{ac}}
  \dnd@ifkey{hp}{\par\noindent\textbf{Hit Points} \dnd@key{hp}\dnd@ifkey{hit_dice}{ (\dnd@key{hit_dice})}}
  \dnd@ifkey{speed}{\par\noindent\textbf{Speed} \dnd@key{speed}}\par}
  \dnd@ifkey{stats}{%
    \dnd@rule
    {\color{DndDarkRed}\expandafter\expandafter\expandafter\dnd@splitstats\csname nopestats\endcsname\dnd@endstats\par}}
  \dnd@rule
  {\color{DndDarkRed}
  \dnd@detail{damage_vulnerabilities}{Damage Vulnerabilities}
  \dnd@detail{damage_resistances}{Damage Resistances}
  \dnd@detail{damage_immunities}{Damage Immunities}
  \dnd@detail{condition_immunities}{Condition Immunities}
  \dnd@detail{senses}{Senses}
  \dnd@detail{languages}{Languages}
  \dnd@detail{cr}{Challenge}\par}
  \dnd@rule
  \dnd@entry{traits}{1}\dnd@entry{traits}{2}\dnd@entry{traits}{3}%
  \dnd@entry{traits}{4}\dnd@entry{traits}{5}\dnd@entry{traits}{6}%
  \dnd@iflist{actions}{\dnd@section{Actions}}%
  \dnd@entry{actions}{1}\dnd@entry{actions}{2}\dnd@entry{actions}{3}\dnd@entry{actions}{4}%
  \dnd@entry{actions}{5}\dnd@entry{actions}{6}\dnd@entry{actions}{7}\dnd@entry{actions}{8}%
  \dnd@iflist{reactions}{\dnd@section{Reactions}}%
  \dnd@entry{reactions}{1}\dnd@entry{reactions}{2}\dnd@entry{reactions}{3}\dnd@entry{reactions}{4}%
  \dnd@iflist{legendary_actions}{\dnd@section{Legendary Actions}%
    \dnd@ifkey{legendary_description}{\par\noindent\dnd@key{legendary_description}}}%
  \dnd@entry{legendary_actions}{1}\dnd@entry{legendary_actions}{2}%
  \dnd@entry{legendary_actions}{3}\dnd@entry{legendary_actions}{4}%
}{\end{dnd@statbox}}
